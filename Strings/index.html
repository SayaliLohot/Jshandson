Count the Heads

        var Count_Occ = (s) => 
        {
          let n = s.length;
          let str = s.split("").sort().join("");
          let ans = "";
          
        for(let i=0;i< n;i++){//for character that need to be repeat
          let count=0;
          for(let j=0; j< n; j++){
            if(str[i]===str[j] && i>j){
              break;
            }
            if(str[i]===str[j]){
              count+=1;
            }
          }
          if(count>1){
            ans+=str[i]+count;
          }
        }
        return ans;
            
        };

        var Reverse_String = (str) => 
        {
           var newString = "";
