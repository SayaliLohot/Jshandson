Find Grades

const findGrades = (a) => {
        //a represents marks like f(a) ...values of a
         var grade = 0;
        if(a<= 10){
          grade = 1;
          //grade E : 1
        }
        if(a>=11 && a<=20)
        {
            grade = 2;
          //grade D : 2
        }
        if(a>=21 && a<=30)
        {
            grade = 3;
          //grade C :3
        }
        if(a>=31 && a<=40)
        {
            grade = 4;
          //grade B : 4
        }
        if(a>=41 && a<=50)
        {
            grade = 5;
          //grade A : 5
        }
        switch(grade)
        {
          case 1:
          return 'E';
          break;
          case 2:
          return 'D';
          break;
          case 3:
          return 'C';
          break;
          case 4:
          return 'B';
          break;
          case 5:
          return 'A';
        }
        };
