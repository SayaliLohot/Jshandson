Birthday Game 

 
            function Birthday_Game(arr,D ,M) {
              var ways=0;
              for(let i=0; i < arr.length-M+1; i++){
                let sum=0;
                for(let j=i; j < M+i; j++){
                  sum+=arr[j];
                }
                if (sum === D){
                  ways++;
                }
              }
              return ways;
            
            }
